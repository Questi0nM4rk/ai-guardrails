# SPEC-002: Configuration System

## Status: Draft

---

## Overview

Two config files. One global, one per-project. Both TOML. Both optional with
sensible defaults. The global sets the machine-wide posture. The project
customises within that.

```
~/.ai-guardrails/config.toml        ← machine-wide (set once at install)
<project>/.ai-guardrails/config.toml ← per-project (committed to repo)
```

---

## Config Hierarchy

```
Profile defaults (built into binary)
    ↓ overridden by
Machine config (~/.ai-guardrails/config.toml)
    ↓ overridden by
Project config (<project>/.ai-guardrails/config.toml)
    ↓ further narrowed by
Inline allow comments (# ai-guardrails-allow: RULE "reason")
```

Rules can only be **removed** as you go down the hierarchy. A project cannot
re-enable a rule the machine config has turned off.

---

## Profiles

Three built-in profiles. Selected at machine or project level.

| Profile | What runs | Typical use |
|---------|-----------|-------------|
| `strict` | Every rule, zero tolerance | AI-first repos, production |
| `standard` | All rules minus high-noise categories (default) | Most projects |
| `minimal` | Core safety rules only | Legacy codebases being adopted |

Profile definitions are built into the binary as `defaults.ts`. Projects
cannot define custom profiles — that's a future enterprise feature.

## Machine Config Schema

```toml
# ~/.ai-guardrails/config.toml
# Generated by: ai-guardrails install
# Edit this to set your machine-wide defaults.

profile = "standard"   # "strict" | "standard" | "minimal"

# Rules turned OFF for ALL projects on this machine.
# Use sparingly — per-project is usually more appropriate.
[[ignore]]
rule = "ruff/D"
reason = "No docstring enforcement on this machine"
```

### Zod schema

```typescript
const IgnoreEntrySchema = z.object({
  rule: z.string().regex(/^[\w-]+\/[\w\-\.]+$/, "Format: linter/RULE_CODE"),
  reason: z.string().min(1, "Reason is required"),
});

const MachineConfigSchema = z.object({
  profile: z.enum(["strict", "standard", "minimal"]).default("standard"),
  ignore: z.array(IgnoreEntrySchema).default([]),
});

export type MachineConfig = z.infer<typeof MachineConfigSchema>;
```

## Project Config Schema

```toml
# <project>/.ai-guardrails/config.toml
# Generated by: ai-guardrails init
# Committed to repo. Controls lint posture for this project.

profile = "standard"   # Optional: inherits from machine config if omitted

# === RULE CONFIGURATION ===
# These values feed directly into generated tool configs (ruff.toml, etc.)
[config]
line_length = 88           # → ruff.toml line-length, .editorconfig max_line_length
indent_width = 4           # → ruff.toml indent-width, .editorconfig indent_size
python_version = "3.11"   # → ruff.toml target-version, mypy python_version

# === PROJECT-WIDE RULE IGNORES ===
# Rules turned OFF for this entire project. Must have a reason.
# Use when a rule doesn't fit the project's design (not just "it fires a lot").
[[ignore]]
rule = "ruff/T201"
reason = "CLI tool — print() is intentional"

[[ignore]]
rule = "ruff/ANN"
reason = "mypy handles type checking; ruff ANN rules are redundant"

# === GLOB-PATTERN ALLOWS ===
# Rule turned off for a specific file pattern. Must have a reason.
# Use when a structural pattern legitimately triggers a rule across many files.
[[allow]]
rule = "ruff/ARG002"
glob = "tests/**/*.py"
reason = "pytest fixtures inject args by signature"

[[allow]]
rule = "ruff/PLR2004"
glob = "tests/**/*.py"
reason = "Magic values in test assertions are self-documenting"

[[allow]]
rule = "ruff/F401"
glob = "**/__init__.py"
reason = "Re-exports in __init__.py are intentional"
```

### Zod schema

```typescript
const ConfigValuesSchema = z.object({
  line_length: z.number().int().min(60).max(200).default(88),
  indent_width: z.number().int().oneOf([2, 4]).default(4),
  python_version: z.string().regex(/^\d+\.\d+$/).optional(),
}).passthrough(); // allow unknown keys for future linter config

const AllowEntrySchema = z.object({
  rule: z.string().regex(/^[\w-]+\/[\w\-\.]+$/),
  glob: z.string().min(1),
  reason: z.string().min(1),
});

const ProjectConfigSchema = z.object({
  profile: z.enum(["strict", "standard", "minimal"]).optional(),
  config: ConfigValuesSchema.default({}),
  ignore: z.array(IgnoreEntrySchema).default([]),
  allow: z.array(AllowEntrySchema).default([]),
});

export type ProjectConfig = z.infer<typeof ProjectConfigSchema>;
```

## ResolvedConfig

The merged result passed to pipelines, steps, runners, and generators. Computed
once at pipeline startup by `config/loader.ts`.

```typescript
export interface ResolvedConfig {
  profile: "strict" | "standard" | "minimal";

  /** Merged ignore list (machine + project), deduped by rule */
  ignore: ReadonlyArray<{ rule: string; reason: string }>;

  /** Project glob-pattern allows */
  allow: ReadonlyArray<{ rule: string; glob: string; reason: string }>;

  /** Merged config values */
  values: {
    line_length: number;
    indent_width: number;
    python_version?: string;
    [key: string]: unknown;
  };

  /** Convenience: set of all globally ignored rule codes */
  ignoredRules: ReadonlySet<string>;

  /** Check if a rule is allowed for a specific file path */
  isAllowed(rule: string, filePath: string): boolean;
}
```

### `isAllowed` logic

```typescript
isAllowed(rule, filePath) {
  if (this.ignoredRules.has(rule)) return true;
  return this.allow.some(
    (entry) => entry.rule === rule && minimatch(filePath, entry.glob)
  );
}
```

## Config Loading (`config/loader.ts`)

```typescript
export async function loadConfig(projectDir: string): Promise<ResolvedConfig> {
  const machineRaw = await readTomlSafe(machineConfigPath());
  const projectRaw = await readTomlSafe(join(projectDir, ".ai-guardrails/config.toml"));

  const machine = MachineConfigSchema.parse(machineRaw ?? {});
  const project = ProjectConfigSchema.parse(projectRaw ?? {});

  const profile = project.profile ?? machine.profile;
  const profileDefaults = PROFILE_DEFAULTS[profile];

  // Merge ignores: profile defaults → machine → project (deduped by rule)
  const ignoreMap = new Map<string, string>();
  for (const entry of [...profileDefaults.ignore, ...machine.ignore, ...project.ignore]) {
    ignoreMap.set(entry.rule, entry.reason);
  }

  const ignore = Array.from(ignoreMap.entries()).map(([rule, reason]) => ({ rule, reason }));
  const ignoredRules = new Set(ignore.map((e) => e.rule));

  const values = { ...profileDefaults.values, ...machine.values, ...project.config };

  return {
    profile,
    ignore,
    allow: project.allow,
    values,
    ignoredRules,
    isAllowed(rule, filePath) {
      if (ignoredRules.has(rule)) return true;
      return project.allow.some(
        (entry) => entry.rule === rule && minimatch(filePath, entry.glob)
      );
    },
  };
}
```

## Inline Allow Comments

Parsed at `check` time from source files. Not stored or persisted. Suppresses
a single line.

### Valid syntax

```python
x = long_url  # ai-guardrails-allow: ruff/E501 "URL cannot be shortened"
```

```typescript
const x = longThing; // ai-guardrails-allow: biome/noExplicitAny "external API type"
```

```html
<!-- ai-guardrails-allow: markdownlint/MD013 "table cell content" -->
```

```lua
local x = val -- ai-guardrails-allow: luacheck/W311 "loop variable shadowing"
```

### Rules

1. **Reason is mandatory.** A quoted string must follow the rule code.
   `# ai-guardrails-allow: E501` with no reason is itself flagged as `AI001`.
2. **Multiple rules.** `# ai-guardrails-allow: E501, E502 "both apply here"`
3. **Rule format.** `linter/RULE_CODE`. The linter prefix is optional in allow
   comments (bare `E501` matches any linter's E501), but prefixed is preferred.
4. **No persisting.** Inline allows are ephemeral — checked at runtime, not
   stored in any registry. The source code IS the record.

### Parser (`hooks/allow-comment.ts`)

```typescript
export interface ParsedAllow {
  rules: readonly string[];  // ["ruff/E501", "ruff/E502"]
  reason: string;
}

/** Parse a valid allow comment. Returns null if not an allow comment or missing reason. */
export function parseAllowComment(line: string): ParsedAllow | null { ... }

/** Parse a bare allow comment (missing reason). Used to emit AI001. */
export function parseBareAllowComment(line: string): readonly string[] | null { ... }
```

## Migration from v1 `.guardrails-exceptions.toml`

The old file had `[global.ruff]` sections for machine-wide ignores and
`[[file_exceptions]]` for glob-pattern exceptions. The new format separates
these cleanly into `[[ignore]]` and `[[allow]]`.

`ai-guardrails init --upgrade` should detect a `v1` file and offer to migrate.
Migration is out of MVP scope but the schema version field enables it:

```toml
# v1 — detected by schema_version = 1
schema_version = 1
```

```toml
# v2 — no version field (or explicit version = 2)
profile = "standard"
```
